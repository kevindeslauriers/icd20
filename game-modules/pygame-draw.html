Pygame Draw Functions

Function Description Parameters Example
pygame.draw.rect() Draws a rectangle on a surface surface, color, rect, width=0 pygame.draw.rect(screen, (255, 0, 0), pygame.Rect(100, 100, 50, 50))
pygame.draw.circle() Draws a circle on a surface surface, color, center, radius, width=0 pygame.draw.circle(screen, (0, 255, 0), (400, 300), 30)
pygame.draw.line() Draws a line on a surface surface, color, start_pos, end_pos, width=1 pygame.draw.line(screen, (255, 255, 255), (0, 0), (800, 600), 2)
pygame.draw.polygon() Draws a polygon on a surface surface, color, pointlist, width=0 pygame.draw.polygon(screen, (255, 255, 0), [(200, 200), (300, 300), (400, 200)])
pygame.draw.arc() Draws an arc on a surface surface, color, rect, start_angle, stop_angle, width=1 pygame.draw.arc(screen, (255, 0, 255), pygame.Rect(100, 100, 200, 200), 0, 3.14, 2)
pygame.draw.ellipse() Draws an ellipse on a surface surface, color, rect, width=0 pygame.draw.ellipse(screen, (0, 0, 255), pygame.Rect(200, 200, 200, 100))
pygame.draw.aaline() Draws an anti-aliased line on a surface surface, color, start_pos, end_pos, blend=1 pygame.draw.aaline(screen, (255, 255, 255), (0, 0), (800, 600))
pygame.draw.aalines() Draws a series of anti-aliased lines on a surface surface, color, closed, pointlist, blend=1 pygame.draw.aalines(screen, (255, 255, 255), True, [(100, 100), (200, 200), (300, 100)])
pygame.draw.lines() Draws a series of connected lines on a surface surface, color, closed, pointlist, width=1 pygame.draw.lines(screen, (255, 255, 0), False, [(100, 100), (200, 200), (300, 100)], 2)
